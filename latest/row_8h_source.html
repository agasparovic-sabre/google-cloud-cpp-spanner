175 // A helper metafunction that promots some C++ literal types to the types
176 // required by Cloud Spanner. For example, a literal 42 is of type int, but
177 // this will promote it to type std::int64_t. Similarly, a C++ string literal
178 // will be "promoted" to type std::string.
179 template <typename T>
180 constexpr T PromoteLiteralImpl(T);
181 constexpr std::string PromoteLiteralImpl(char const*);
182 constexpr std::int64_t PromoteLiteralImpl(int);
183 template <typename T>
184 using PromoteLiteral = decltype(PromoteLiteralImpl(std::declval<T>()));
185 } // namespace internal

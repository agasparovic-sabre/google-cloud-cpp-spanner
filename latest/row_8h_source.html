211 // A helper metafunction that promots some C++ literal types to the types
212 // required by Cloud Spanner. For example, a literal 42 is of type int, but
213 // this will promote it to type std::int64_t. Similarly, a C++ string literal
214 // will be "promoted" to type std::string.
215 template <typename T>
216 constexpr T PromoteLiteralImpl(T);
217 constexpr std::string PromoteLiteralImpl(char const*);
218 constexpr std::int64_t PromoteLiteralImpl(int);
219 template <typename T>
220 using PromoteLiteral = decltype(PromoteLiteralImpl(std::declval<T>()));

222 // A helper functor to be used with `internal::ForEach` to iterate the columns
223 // of a `Row<Ts...>` in the ParseRow() function.
224 struct ExtractValue {
225  Status& status;
226  template <typename T, typename It>
227  void operator()(T& t, It& it) const {
228  auto x = it++->template get<T>();
229  if (!x) {
230  status = std::move(x).status();
231  } else {
232  t = *std::move(x);
233  }
234  }

292 // A helper metafunction that promots some C++ literal types to the types
293 // required by Cloud Spanner. For example, a literal 42 is of type int, but
294 // this will promote it to type std::int64_t. Similarly, a C++ string literal
295 // will be "promoted" to type std::string.
296 template <typename T>
297 T PromoteLiteralImpl(T);
298 std::string PromoteLiteralImpl(char const*);
299 std::int64_t PromoteLiteralImpl(int);
300 template <typename T>
301 using PromoteLiteral = decltype(PromoteLiteralImpl(std::declval<T>()));

303 // A helper functor to be used with `internal::ForEach` to iterate the columns
304 // of a `Row<Ts...>` in the ParseRow() function.
305 struct ExtractValue {
306  Status& status;
307  template <typename T, typename It>
308  void operator()(T& t, It& it) const {
309  auto x = it++->template get<T>();
310  if (!x) {
311  status = std::move(x).status();
312  } else {
313  t = *std::move(x);
314  }
315  }
316  template <typename It>
317  void operator()(Value& v, It& it) const {
318  v = *it++;
319  }

359 template <typename... Ts>
360 StatusOr<Row<internal::PromoteLiteral<Ts>...>> ParseRow(
361  std::array<Value, sizeof...(Ts)> const& array) {
362  Row<internal::PromoteLiteral<Ts>...> row;
363  auto it = array.begin();
364  Status status;
365  internal::ForEach(row, internal::ExtractValue{status}, it);
366  if (!status.ok()) return status;

28 std::string RandomDatabaseName(
29  google::cloud::internal::DefaultPRNG& generator) {
30  // A database ID must be between 2 and 30 characters, fitting the regular
31  // expression `[a-z][a-z0-9_\-]*[a-z0-9]`
32  int max_size = 30;
33  std::string const prefix = "db-";
34  auto size = static_cast<int>(max_size - 1 - prefix.size());
35  return prefix +
36  google::cloud::internal::Sample(
37  generator, size, "abcdefghijlkmnopqrstuvwxyz012345689_-") +
38  google::cloud::internal::Sample(generator, 1,
39  "abcdefghijlkmnopqrstuvwxyz");
40 }

298 void ReadOnlyTransaction(google::cloud::spanner::Client client) {
299  namespace spanner = google::cloud::spanner;
300  auto read_only = spanner::MakeReadOnlyTransaction();
301 
302  spanner::SqlStatement select(
303  "SELECT SingerId, AlbumId, AlbumTitle FROM Albums");
304 
305  // Read#1.
306  auto read1 = client.ExecuteSql(read_only, select);
307  std::cout << "Read 1 results\n";
308  for (auto row : read1->Rows<std::int64_t, std::int64_t, std::string>()) {
309  if (!row) {
310  throw std::runtime_error(row.status().message());
311  }
312  std::cout << "SingerId: " << row->get<0>() << " AlbumId: " << row->get<1>()
313  << " AlbumTitle: " << row->get<2>() << "\n";
314  }
315  // Read#2. Even if changes occur in-between the reads the transaction ensures
316  // that Read #1 and Read #2 return the same data.
317  auto read2 = client.ExecuteSql(read_only, select);
318  std::cout << "Read 2 results\n";
319  for (auto row : read2->Rows<std::int64_t, std::int64_t, std::string>()) {
320  if (!row) {
321  throw std::runtime_error(row.status().message());
322  }
323  std::cout << "SingerId: " << row->get<0>() << " AlbumId: " << row->get<1>()
324  << " AlbumTitle: " << row->get<2>() << "\n";
325  }

330 void ReadWriteTransaction(google::cloud::spanner::Client client) {
331  namespace spanner = google::cloud::spanner;
332  auto txn = spanner::MakeReadWriteTransaction();
333 
334  auto get_current_budget = [&txn, &client](std::int64_t singer_id,
335  std::int64_t album_id) {
336  auto key = spanner::KeySetBuilder<spanner::Row<std::int64_t, std::int64_t>>(
337  spanner::MakeRow(singer_id, album_id))
338  .Build();
339  auto read = client.Read(txn, "Albums", std::move(key), {"MarketingBudget"});
340  if (!read) {
341  throw std::runtime_error(read.status().message());
342  }
343  for (auto row : read->Rows<std::int64_t>()) {
344  if (!row) {
345  throw std::runtime_error(read.status().message());
346  }
347  // We expect at most one result from the `Read()` request. Return the
348  // first one.
349  return row->get<0>();
350  }
351  throw std::runtime_error("Key not found (" + std::to_string(singer_id) +
352  "," + std::to_string(album_id) + ")");
353  };
354 
355  auto b1 = get_current_budget(1, 1);
356  auto b2 = get_current_budget(2, 2);
357  std::int64_t transfer_amount = 200000;
358 
359  auto commit_result = client.Commit(
360  txn, {spanner::UpdateMutationBuilder(
361  "Albums", {"SingerId", "AlbumId", "MarketingBudget"})
362  .EmplaceRow(1, 1, b1 + transfer_amount)
363  .EmplaceRow(2, 2, b2 - transfer_amount)
364  .Build()});
365 
366  if (!commit_result) {
367  throw std::runtime_error(commit_result.status().message());
368  }
369  std::cout << "Transfer was successful [spanner_read_write_transaction]\n";
370 }

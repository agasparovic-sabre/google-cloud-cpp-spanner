25 std::string RandomDatabaseName(
26  google::cloud::internal::DefaultPRNG& generator) {
27  // A database ID must be between 2 and 30 characters, fitting the regular
28  // expression `[a-z][a-z0-9_\-]*[a-z0-9]`
29  int max_size = 30;
30  std::string const prefix = "db-";
31  auto size = static_cast<int>(max_size - 1 - prefix.size());
32  return prefix +
33  google::cloud::internal::Sample(
34  generator, size, "abcdefghijlkmnopqrstuvwxyz012345689_-") +
35  google::cloud::internal::Sample(generator, 1,
36  "abcdefghijlkmnopqrstuvwxyz");
37 }

57  FirstName STRING(1024),
58  LastName STRING(1024),
59  SingerInfo BYTES(MAX)
60  ) PRIMARY KEY (SingerId))""",
61  R"""(
62  CREATE TABLE Albums (
63  SingerId INT64 NOT NULL,
64  AlbumId INT64 NOT NULL,
65  AlbumTitle STRING(MAX)
66  ) PRIMARY KEY (SingerId, AlbumId),

68  StatusOr<google::spanner::admin::database::v1::Database> db = future.get();
69  if (!db) {

147 void InsertData(google::cloud::spanner::Client client) {

149  auto insert_singers = spanner::InsertMutationBuilder(
150  "Singers", {"SingerId", "FirstName", "LastName"})
151  .EmplaceRow(1, "Marc", "Richards")
152  .EmplaceRow(2, "Catalina", "Smith")
153  .EmplaceRow(3, "Alice", "Trentor")
154  .EmplaceRow(4, "Lea", "Martin")
155  .EmplaceRow(5, "David", "Lomond")
156  .Build();
157 
158  auto insert_albums = spanner::InsertMutationBuilder(
159  "Albums", {"SingerId", "AlbumId", "AlbumTitle"})
160  .EmplaceRow(1, 1, "Total Junk")
161  .EmplaceRow(1, 2, "Go, Go, Go")
162  .EmplaceRow(2, 1, "Green")
163  .EmplaceRow(2, 2, "Forever Hold Your Peace")
164  .EmplaceRow(2, 3, "Terrified")
165  .Build();
166 
167  auto commit_result =
168  client.Commit(spanner::MakeReadWriteTransaction(),
169  {std::move(insert_singers), std::move(insert_albums)});
170  if (!commit_result) {
171  throw std::runtime_error(commit_result.status().message());
172  }
173  std::cout << "Insert was successful\n";

253 int RunOneCommand(std::vector<std::string> argv) {
254  using CommandType = std::function<void(std::vector<std::string> const&)>;
255 
256  std::map<std::string, CommandType> commands = {
257  {"create-database", &CreateDatabase},
258  {"add-column", &AddColumn},
259  {"query-with-struct", &QueryWithStructCommand},
260  {"drop-database", &DropDatabase},
261  {"insert-data", &InsertDataCommand},
262  {"dml-standard-insert", &DmlStandardInsertCommand},
263  {"dml-standard-update", &DmlStandardUpdateCommand},
264  };
265 
266  static std::string usage_msg = [&argv, &commands] {
267  auto last_slash = std::string(argv[0]).find_last_of("/\\");
268  auto program = argv[0].substr(last_slash + 1);
269  std::string usage;
270  usage += "Usage: ";
271  usage += program;
272  usage += " <command> [arguments]\n\n";
273  usage += "Commands:\n";
274  for (auto&& kv : commands) {
275  try {
276  kv.second({});
277  } catch (std::exception const& ex) {
278  usage += " ";
279  usage += ex.what();
280  usage += "\n";
281  }
282  }
283  return usage;
284  }();
285 
286  if (argv.size() < 2) {
287  std::cerr << "Missing command argument\n" << usage_msg << "\n";
288  return 1;
289  }
290  std::string command_name = argv[1];
291  argv.erase(argv.begin()); // remove the program name from the list.
292  argv.erase(argv.begin()); // remove the command name from the list.
293 
294  auto command = commands.find(command_name);
295  if (commands.end() == command) {
296  std::cerr << "Unknown command " << command_name << "\n"
297  << usage_msg << "\n";
298  return 1;
299  }
300 
301  // Run the command.
302  command->second(argv);
303  return 0;
304 }

306 void RunAll() {
307  auto project_id =
308  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value_or("");
309  auto instance_id =
310  google::cloud::internal::GetEnv("GOOGLE_CLOUD_CPP_SPANNER_INSTANCE")
311  .value_or("");
312 
313  if (project_id.empty()) {
314  throw std::runtime_error("GOOGLE_CLOUD_PROJECT is not set or is empty");
315  }
316  if (instance_id.empty()) {
317  throw std::runtime_error(
318  "GOOGLE_CLOUD_CPP_SPANNER_INSTANCE is not set or is empty");
319  }
320  auto generator = google::cloud::internal::MakeDefaultPRNG();
321  std::string database_id = RandomDatabaseName(generator);
322 
323  RunOneCommand({"", "create-database", project_id, instance_id, database_id});
324  RunOneCommand({"", "add-column", project_id, instance_id, database_id});
325 
326  namespace spanner = google::cloud::spanner;
327  spanner::Database db(project_id, instance_id, database_id);
328  google::cloud::spanner::Client client(
329  google::cloud::spanner::MakeConnection(db));

333  QueryWithStructCommand({project_id, instance_id, database_id});

337  RunOneCommand({"", "drop-database", project_id, instance_id, database_id});

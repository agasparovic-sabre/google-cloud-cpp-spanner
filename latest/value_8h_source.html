359  static google::spanner::v1::Type MakeTypeProto(std::vector<T> const& v) {
360  google::spanner::v1::Type t;
361  t.set_code(google::spanner::v1::TypeCode::ARRAY);
362  *t.mutable_array_element_type() = MakeTypeProto(v.empty() ? T{} : v[0]);
363  // Checks that vector elements have exactly the same proto Type, which
364  // includes field names. This is documented UB.
365  for (auto const& e : v) {
366  if (!google::protobuf::util::MessageDifferencer::Equals(
367  MakeTypeProto(e), t.array_element_type())) {
368  google::cloud::internal::ThrowInvalidArgument("Mismatched types");
369  }
370  }
371  return t;
372  }

510  // A helper to iterate the elements of the tuple, calling the given functor
511  // `f` with each tuple element and any optional `args`. Typically `F` should
512  // be a functor type with a templated operator() so that it can handle the
513  // various types in the tuple.
514  template <std::size_t I = 0, typename Tup, typename F, typename... Args>
515  static typename std::enable_if<
516  (I < std::tuple_size<typename std::decay<Tup>::type>::value), void>::type
517  IterateTuple(Tup&& tup, F&& f, Args&&... args) {
518  auto&& e = std::get<I>(std::forward<Tup>(tup));
519  std::forward<F>(f)(std::forward<decltype(e)>(e),
520  std::forward<Args>(args)...);
521  IterateTuple<I + 1>(std::forward<Tup>(tup), std::forward<F>(f),
522  std::forward<Args>(args)...);
523  }
524  template <std::size_t I = 0, typename Tup, typename F, typename... Args>
525  static typename std::enable_if<
526  I == std::tuple_size<typename std::decay<Tup>::type>::value, void>::type
527  IterateTuple(Tup&&, F&&, Args&&...) {}

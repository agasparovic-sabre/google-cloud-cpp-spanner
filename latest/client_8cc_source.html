125 StatusOr<CommitResult> RunTransaction(
126  Client client, Transaction::ReadWriteOptions const& opts,
127  std::function<StatusOr<Mutations>(Client, Transaction)> const& f) {
128  Transaction txn = MakeReadWriteTransaction(opts);
129  StatusOr<Mutations> mutations;
130 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
131  try {
132 #endif
133  mutations = f(client, txn);
134 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
135  } catch (...) {
136  auto status = client.Rollback(txn);
137  if (!status.ok()) {
138  GCP_LOG(WARNING) << "Rollback() failure in RunTransaction(): "
139  << status.message();
140  }
141  throw;
142  }
143 #endif
144  if (!mutations) {
145  auto status = client.Rollback(txn);
146  if (!status.ok()) {
147  GCP_LOG(WARNING) << "Rollback() failure in RunTransaction(): "
148  << status.message();
149  }
150  return mutations.status();
151  }
152  // TODO(#357): Automatically retry if Commit() aborts. Note: it is
153  // not a good idea to simply cap the number of retries. Instead, it
154  // is better to limit the total amount of wall time spent retrying.
155  return client.Commit(txn, *mutations);
156 }

122 StatusOr<CommitResult> RunTransaction(
123  Client client, Transaction::ReadWriteOptions const& opts,
124  std::function<TransactionAction(Client, Transaction)> const& f) {
125  Transaction txn = MakeReadWriteTransaction(opts);
126  TransactionAction action;
127 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
128  try {
129 #endif
130  action = f(client, txn);
131 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
132  } catch (...) {
133  client.Rollback(txn); // ignore error
134  throw;
135  }
136 #endif
137  switch (action.action) {
138  case TransactionAction::kCommit: {
139  // TODO(#357): Automatically retry if Commit() aborts. Note: it is
140  // not a good idea to simply cap the number of retries. Instead, it
141  // is better to limit the total amount of wall time spent retrying.
142  return client.Commit(txn, action.mutations);
143  }
144  case TransactionAction::kRollback: {
145  auto status = client.Rollback(txn);
146  if (!status.ok()) return status;
147  return CommitResult{}; // TODO(#357): What should this return?
148  }
149  }
150  return Status(StatusCode::kInvalidArgument, "bad TransactionAction");
151 }

209 class KeySet {
210  public:
214  static KeySet All() {
215  KeySet ks;
216  ks.proto_.set_all(true);
217  return ks;
218  }
219 
223  KeySet() = default;
224 
230  template <typename RowType>
231  explicit KeySet(KeySetBuilder<RowType> const& builder) {
232  for (auto& key : builder.keys()) {
233  Append(proto_.add_keys(), std::move(key).values());
234  }
235  for (auto& range : builder.key_ranges()) {
236  auto* kr = proto_.add_ranges();
237  auto* start = range.start().IsClosed() ? kr->mutable_start_closed()
238  : kr->mutable_start_open();
239  Append(start, std::move(range).start().key().values());
240  auto* end = range.end().IsClosed() ? kr->mutable_end_closed()
241  : kr->mutable_end_open();
242  Append(end, std::move(range).end().key().values());
243  }
244  }
245 
246  private:
247  friend ::google::spanner::v1::KeySet internal::ToProto(KeySet keyset);
248 
249  template <std::size_t N>
250  static void Append(google::protobuf::ListValue* lv, std::array<Value, N> a) {
251  for (auto& v : a) {
252  *lv->add_values() = internal::ToProto(std::move(v)).second;
253  }
254  }

257 };

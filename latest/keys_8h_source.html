361  for (auto& key : keys_) {
362  Append(ks.add_keys(), std::move(key).values());
363  }
364  for (auto& range : key_ranges_) {
365  auto* kr = ks.add_ranges();
366  auto* start = range.start().IsClosed() ? kr->mutable_start_closed()
367  : kr->mutable_start_open();
368  Append(start, std::move(range).start().key().values());
369  auto* end = range.end().IsClosed() ? kr->mutable_end_closed()
370  : kr->mutable_end_open();
371  Append(end, std::move(range).end().key().values());
372  }

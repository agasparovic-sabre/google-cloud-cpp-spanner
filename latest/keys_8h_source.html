224 class KeySet {
225  public:
229  static KeySet All() {
230  KeySet ks;
231  ks.proto_.set_all(true);
232  return ks;
233  }

238  KeySet() = default;

245  template <typename RowType>
246  explicit KeySet(KeySetBuilder<RowType> const& builder) {
247  for (auto& key : builder.keys()) {
248  Append(proto_.add_keys(), std::move(key).values());
249  }
250  for (auto& range : builder.key_ranges()) {
251  auto* kr = proto_.add_ranges();
252  auto* start = range.start().IsClosed() ? kr->mutable_start_closed()
253  : kr->mutable_start_open();
254  Append(start, std::move(range).start().key().values());
255  auto* end = range.end().IsClosed() ? kr->mutable_end_closed()
256  : kr->mutable_end_open();
257  Append(end, std::move(range).end().key().values());
258  }
259  }
260 
261  private:
262  friend ::google::spanner::v1::KeySet internal::ToProto(KeySet keyset);
263 
264  template <std::size_t N>
265  static void Append(google::protobuf::ListValue* lv, std::array<Value, N> a) {
266  for (auto& v : a) {
267  *lv->add_values() = internal::ToProto(std::move(v)).second;
268  }
269  }

272 };
